     color-scheme: dark;

   ```tsx
   import type { Metadata } from 'next'
   import { Playfair_Display, DM_Sans, DM_Mono } from 'next/font/google'
   import './globals.css'

   const playfair = Playfair_Display({
     subsets: ['latin'],
     weight: ['400', '700'],
     variable: '--font-display',
     display: 'swap',
   })

   const dmSans = DM_Sans({
     subsets: ['latin'],
     variable: '--font-body',
     display: 'swap',
   })

   const dmMono = DM_Mono({
     subsets: ['latin'],
     weight: ['300', '400', '500'],
     variable: '--font-mono',
     display: 'swap',
   })

   export const metadata: Metadata = {
     title: 'Vidit Dugar — Investor. Operator. Builder.',
     description: 'Early-stage consumer investor in India. Formerly Nomura London, Bombay Shaving Company, The Barbershop Fund.',
   }

   export default function RootLayout({
     children,
   }: {
     children: React.ReactNode
   }) {
     return (
       <html lang="en" className={`${playfair.variable} ${dmSans.variable} ${dmMono.variable}`}>
         <body className="font-body antialiased">
           {children}
         </body>
       </html>
     )
   }
   ```

   Note: Playfair Display is NOT a variable font — explicit weights (400, 700) are required. DM Sans and DM Mono are variable fonts loaded with default variable axes.
