**Step 2: Create `components/about/AboutAnimations.tsx`** — a `'use client'` component that receives `prose` (string) and `pullQuote` (string) as props. Uses `m.div` with `whileInView` + `viewport={{ once: true }}` for fade-in. Structure:

```tsx
'use client'
import { m } from 'framer-motion'

interface AboutAnimationsProps {
  prose: string
  pullQuote: string
}

export function AboutAnimations({ prose, pullQuote }: AboutAnimationsProps) {
  // Split prose into paragraphs by double newline
  const paragraphs = prose.split('\n\n')

  return (
    <div>
      <m.div initial={{ opacity: 0, y: 16 }} whileInView={{ opacity: 1, y: 0 }} viewport={{ once: true }} transition={{ duration: 0.6, ease: 'easeOut' }}>
        <h2 className="font-display text-3xl md:text-4xl font-bold text-foreground mb-10">The Story</h2>
      </m.div>

      {paragraphs.map((p, i) => (
        <m.p key={i} initial={{ opacity: 0, y: 12 }} whileInView={{ opacity: 1, y: 0 }} viewport={{ once: true }} transition={{ duration: 0.5, ease: 'easeOut', delay: i * 0.08 }} className="font-body text-base text-muted leading-relaxed mb-6">
          {p}
        </m.p>
      ))}

      {/* Pull quote — NARR-02 */}
      <m.blockquote initial={{ opacity: 0, y: 12 }} whileInView={{ opacity: 1, y: 0 }} viewport={{ once: true }} transition={{ duration: 0.6, ease: 'easeOut', delay: 0.1 }} className="my-12 pl-6 border-l-2 border-accent">
        <p className="font-display text-2xl md:text-3xl italic text-foreground leading-snug">
          "{pullQuote}"
        </p>
      </m.blockquote>
    </div>
  )
}
```

```tsx
'use client'
import { m } from 'framer-motion'
import type { PhilosophyPillar } from '@/lib/types'

export function PhilosophyAnimations({ pillars }: { pillars: PhilosophyPillar[] }) {

    <div>
      <m.div initial={{ opacity: 0, y: 16 }} whileInView={{ opacity: 1, y: 0 }} viewport={{ once: true }} transition={{ duration: 0.6, ease: 'easeOut' }}>

      <div className="space-y-16">
        {pillars.map((pillar, i) => (
          <m.div
            key={pillar.number}
            initial={{ opacity: 0, y: 16 }}
            whileInView={{ opacity: 1, y: 0 }}
            viewport={{ once: true }}
            transition={{ duration: 0.6, ease: 'easeOut', delay: i * 0.1 }}
          >
            <span className="font-mono text-4xl text-accent/30 block mb-2">
              {pillar.number}
            </span>
            <h3 className="font-display text-xl font-bold text-foreground mb-4">
              {pillar.title}
            </h3>
            <p className="font-body text-base text-muted leading-relaxed whitespace-pre-line">
              {pillar.body}
            </p>
          </m.div>
        ))}
      </div>
    </div>
  )
}
```

Key details:
- Pillar number: `font-mono text-4xl text-accent/30` — large, muted gold numeral
- Pillar title: `font-display text-xl font-bold text-foreground` — Playfair Display bold
- Pillar body: `font-body text-base text-muted leading-relaxed whitespace-pre-line` — whitespace-pre-line preserves the `\n\n` paragraph breaks in the body text
- Stagger: `delay: i * 0.1` so pillars reveal sequentially as user scrolls
- Uses `m.div` (NOT `motion.div`) — LazyMotion requires `m.*`
